SELECT Cust.CustomerID,
       OrderHeader.CustomerID,
       OrderHeader.SalesOrderID,
       OrderHeader.Status
FROM Sales.Customer Cust 
	INNER JOIN Sales.SalesOrderHeader OrderHeader
		ON Cust.CustomerID = OrderHeader.CustomerID
FOR XML AUTO;
